flow = group c by int ( c.time /10000000) in mem_swap_free where c.time > <<start:0>> and c.time < <<end:0>> and ( <<nodes:none>> )
 var sum
 var count
 var time
 var max
   sum = sum + c.value
   count = count + 1
   time = int ( c.time /10000000)
   if max < c.value
     max = c.value
   end
final
 yield time => int(time*10000), avg => sum / count, max => max
end
